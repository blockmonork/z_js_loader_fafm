fafm.encode/decode base64

Texto a ser codificado/decodificado com base64 usando fafm.encode

function codificar()
{
	var divId = 'teste_codifica_decodifica';
	var texto = $('#'+divId).html();
	var x = fafm.encode(texto);
	fafm.id(divId, x);
}
codificar 
function decodificar()
{
	var divId = 'teste_codifica_decodifica';
	var texto = $('#'+divId).html();
	var x = fafm.decode(texto);
	fafm.id(divId, x);
}
