function send_hash_to_ajax(_href)
{
	v = fafm.get_hash_from_url(_href);
	fafm.ajax('ajax_hash', 'index.php?ajaxByHash='+v, 'ajax_hash_loader');
}
$(document).ready(function(){
	$('.ajaxHash').click(function(e){
		e.preventDefault();
		send_hash_to_ajax( $(this).attr('href') );
	});
});
