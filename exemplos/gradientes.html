Plugin Gradient

$('#elementoID').gradient(defaultOptions{
	mode 					: 'linear', // ou radial
	degrees					: false, // se true, informar o valor em direction. ex:direction: '45deg'
	repeating 				: false,
	direction 				: 'top', 
	colors 					: 'white, black',
	radialCenter			: '', // padrao: centro ou 50%, 50% ou px-width, px-height 
	radialSize				: 4, // quanto menor o valor, mais longe da borda estará (valores de 1 a 4)
	keepAtualElementStyle 	: true, // setar para false se não desejar que o plugin 
									mantenha o style do elemento setado (ex div style=qqerCore)
	});
a referência pode ser:
$(elemento) > $('p').gradient();
$(classe) > $('.classeQualquer').gradient();
$(elementoID) > $('#elementoIdX').gradient();
