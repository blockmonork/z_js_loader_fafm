Get set attribute

function teste_get_set_attr(id)
{
	w = fafm.get_attribute(id, 'clientWidth') || fafm.get_attribute(id, 'width');
	texto = 'largura('+w+')';
	fafm.id('get_set_attr_resposta', texto);
}
function teste_altera_attr(id)
{
	fafm.set_attribute(id, 'style', 'width:250px; height:250px; color:white; background:red;');
}
function teste_remove_attr(id, at)
{
	fafm.remove_attribute(id, at);
	fafm.id('get_set_attr_resposta', at + ' removido');
}
